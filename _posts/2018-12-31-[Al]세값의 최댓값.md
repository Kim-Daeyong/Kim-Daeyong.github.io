        //세 값의 최댓값.

        Scanner sc = new Scanner(System.in);

        System.out.println("세정수의 최댓값을 입력");
        System.out.println("a의 값 : ");
        int a = sc.nextInt();
        System.out.println("b의 값 : ");
        int b = sc.nextInt();
        System.out.println("c의 값 : ");
        int c = sc.nextInt();

        int max = a;
        if (b > max) max = b;
        if (c > max) max = c;

        System.out.println("최댓값은 " +max+ "입니다.");
